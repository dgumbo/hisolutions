--Install RBENV
1. git clone https://github.com/rbenv/rbenv.git ~/.rbenv
2. echo 'export PATH="$HOME/.rbenv/bin:$PATH"' >> ~/.bash_profile

3. ~/.rbenv/bin/rbenv init
# Load rbenv automatically by appending
# the following to ~/.bash_profile:
# eval "$(rbenv init -)"

4. vi ~/.bash_profile
# *i insert text before cursor, until <Esc> hit
# then eval "$(rbenv init -)"
# then Shift+Z +Z to save and exit

5. which rben
6. which rbenv
7. source ~/.bash_profile

8. git clone git://github.com/dcarley/rbenv-sudo.git ~/.rbenv/plugins/rbenv-sudo
9. git clone https://github.com/rbenv/ruby-build.git ~/.rbenv/plugins/ruby-build

10. sudo yum groupinstall "Development tools" - to install gcc compiler

11. sudo yum install sqlite-devel openssl-devel readline-devel zlib-devel mysql-devel

14. sudo yum install bzip2 autoconf automake libtool bison iconv-devel sqlite-devel mysql-devel
